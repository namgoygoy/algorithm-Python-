## 21876번 **Squarow**

### 문제 설명

숫자 간의 병합을 예상하는 문제

**남은 블록 수 = (원래 총 블록 수) - (사라지는 C 블록 수) - (C가 사라져서 발생하는 병합 횟수)**

## 코드

```jsx
import sys
from collections import defaultdict

n_str = sys.stdin.readline()
n = int(n_str)

if n == 0:
    print("0 0")
else:
    squares = list(map(int, sys.stdin.readline().split()))

    blocks = []
    blocks.append(squares[0])
    
    for i in range(1, n):
        if squares[i] != blocks[-1]:
            blocks.append(squares[i])

    total_block_count = len(blocks)
    
    block_counts = defaultdict(int)
    merge_counts = defaultdict(int)
    all_colors = set()

    for i in range(total_block_count):
        color = blocks[i]
        all_colors.add(color)
        
        block_counts[color] += 1
        
        if 0 < i < total_block_count - 1:
            if blocks[i-1] == blocks[i+1]:
                merge_counts[color] += 1

    best_score = -1
    color_to_remove = -1

    for color in all_colors:
        
        removed_blocks = block_counts[color]
        merged_blocks = merge_counts[color]
        
        current_score = (total_block_count - removed_blocks) - merged_blocks
        
        if current_score > best_score:
            best_score = current_score
            color_to_remove = color
        
        elif current_score == best_score:
            if color_to_remove == -1 or color < color_to_remove:
                 color_to_remove = color

    if color_to_remove == -1:
        color_to_remove = blocks[0]
        best_score = 0 

    print(f"{best_score} {color_to_remove}")
```

## 코드

```jsx
import sys

def solve():
    # 빠른 입력을 위해 sys.stdin.readline 사용
    n_str = sys.stdin.readline()
    if not n_str:
        return  # 입력이 비었으면 종료
    n = int(n_str)
    
    if n == 0:
        print("0 0")
        return

    try:
        a = list(map(int, sys.stdin.readline().split()))
    except EOFError:
        return
    except ValueError:
        return

    # 1. 블록 압축 (연속된 중복 제거)
    # 예: [1, 1, 1, 2, 2, 1] -> [1, 2, 1]
    b = []
    if n > 0:
        b.append(a[0])
        for i in range(1, n):
            if a[i] != b[-1]:
                b.append(a[i])
    
    m = len(b)  # 압축된 배열의 길이 (전체 블록 수)

    # 2. 사전 계산 (블록 수와 병합 수 계산)
    # 문제 제약 조건에서 색깔은 1부터 200,000까지 가능
    # 0번 인덱스는 사용하지 않으므로 200,001 크기로 생성
    MAX_VAL = 200005 
   
    # count[C]: 색깔 C를 가진 블록이 총 몇 개인지
    count = [0] * MAX_VAL
    # merges[C]: C를 제거했을 때 양옆이 같아져서 병합되는 횟수
    merges = [0] * MAX_VAL
    
    unique_colors = set() # 실제 존재하는 색깔만 순회하기 위함

    for j in range(m):
        color = b[j]
        unique_colors.add(color)
        
        # (1) 이 색깔 블록의 수를 1 증가
        count[color] += 1
        
        # (2) 병합(merge)이 일어나는지 확인
        # j가 0(맨 앞)이나 m-1(맨 뒤)이 아니어야 함
        if 0 < j < m - 1:
            # 현재 블록(j)의 양옆(j-1, j+1) 색깔이 같다면
            if b[j-1] == b[j+1]:
                # 현재 색깔(color)을 제거하면 병합이 일어남
                merges[color] += 1

    # 3. 최대 점수 찾기
    max_score = -1
    best_color = -1

    # 모든 색깔(1~200000)을 다 볼 필요 없이,
    # 실제로 입력에 존재했던 색깔들(unique_colors)만 확인
    for color in unique_colors:
        # 점수 계산 공식
        score = (m - count[color]) - merges[color]
        
        # 현재 점수가 최대 점수보다 높으면 갱신
        if score > max_score:
            max_score = score
            best_color = color
        # 만약 최대 점수가 같은 경우,
        # 문제에서 "아무거나 출력"하라고 했으므로
        # 더 작은 색깔을 선호한다면 아래 조건 추가
        # elif score == max_score and color < best_color:
        #     best_color = color
        
        # (예외 처리) 만약 모든 색깔이 점수가 0이라서
        # max_score가 -1로 남아있는 경우를 대비해
        # best_color가 한 번도 할당되지 않았다면 할당
        if best_color == -1:
            best_color = color

    print(f"{max_score} {best_color}")

# 함수 실행
solve()
```
